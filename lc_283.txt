283.Move Zeroes
public void moveZeroes(int[] nums) {
    int length = nums.length;
    for (int i = 0; i < length; i++) {
        if (nums[i] == 0) {
            int j = i;
            while (j < length - 1 && nums[j] == 0) {
                j++;
            }
            int tmp = nums[i];
            nums[i] = nums[j];
            nums[j] = tmp;
        }
    }
}
