1642.Furthest Building You Can Reach
public int furthestBuilding(int[] heights, int bricks, int ladders) {
    for (int i = 1; i < heights.length; i++) {
        int diff = heights[i] - heights[i - 1];
        if (diff > 0) {
            if (diff <= bricks) {
                bricks -= diff;
            } else if (ladders > 0) {
                ladders--;
            } else {
                return i - 1;
            }
        }
    }
    return heights.length - 1;
}
