public int stoneGameVIII(int[] stones) {
    int length = stones.length;
    for (int i = 1; i < length; i++) {
        stones[i] += stones[i - 1];
    }
    int[] dp = new int[length];
    dp[length - 2] = stones[length - 1];
    for (int i = length - 2; i > 0; i--) {
        dp[i - 1] = Math.max(dp[i], stones[i] - dp[i]);
    }
    return dp[0];
}
