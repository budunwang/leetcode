189.Rotate Array
public void rotate(int[] nums, int k) {
    int length = nums.length;
    k = k % length;
    int[] arr = new int[k];
    for (int i = 0; i < k; i++) {
        arr[i] = nums[length - k + i];
    }
    for (int i = 0; i < length - k; i++) {
        nums[length - 1 - i] = nums[length - 1 - i - k];
    }
    for (int i = 0; i < k; i++) {
        nums[i] = arr[i];
    }
}
