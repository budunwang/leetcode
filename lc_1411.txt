public int numOfWays(int n) {
    long col2 = 6;
    long col3 = 6;
    int mod = (int)(1e9 + 7);
    for (int i = 2; i <= n; i++) {
        long tmp = col2; 
        col2 = (2 * col3 + 3 * col2) % mod;
        col3 = (2 * col3 + 2 * tmp) % mod;
    }
    return (int)(col2 + col3) % mod;
}
