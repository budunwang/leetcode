1437.Check If All 1's Are at Least Length K Places Away
public boolean kLengthApart(int[] nums, int k) {
    int last = -1;
    for (int i = 0; i < nums.length; i++) {
        if (nums[i] == 1) {
            if (last != -1 && i - last - 1 < k) {
                return false;                    
            }
            last = i;
        }
    }
    return true;
}
