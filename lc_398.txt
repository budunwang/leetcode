398.Random Pick Index
public int[] arr;
public Random rand;

public Solution(int[] nums) {
    arr = nums;
    rand = new Random();
}

public int pick(int target) {
    int count = 0;
    int idx = 0;
    for (int i = 0; i < arr.length; i++) {
        if (arr[i] == target) {
            count++;
            if (rand.nextInt(count) == 0) {
                idx = i;
            }
        }
    }
    return idx;
}
